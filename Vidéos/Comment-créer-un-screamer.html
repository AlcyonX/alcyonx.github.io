Comment créer un screamer ?

Vidéo

Code

local image = script:FindFirstChildOfClass("Decal")
local sound = script:FindFirstChildOfClass("Sound")
 
script.Parent.Touched:Connect(function(hit)
    if image ~= nil and sound ~= nil then
        local plr
        local gui
        local success, message = pcall(function()
            plr = game.Players:GetPlayerFromCharacter(hit.Parent)
            gui = plr.PlayerGui
        end)
        if success then
            local humanoid = hit.Parent:FindFirstChild("Humanoid")
            humanoid.WalkSpeed = 0
            if gui:FindFirstChild("Scream") then
                return
            end
            local Screen = Instance.new("ScreenGui", gui)
            Screen.Name = "Scream"
            Screen.IgnoreGuiInset = true
            local imagegui = Instance.new("ImageLabel", Screen)
            imagegui.Size =  UDim2.new(1, 0, 1, 0)
            imagegui.Image = image.Texture
            local soundclone = sound:Clone()
            soundclone.Parent = script.Parent
            soundclone:Play()
            wait(soundclone.TimeLength)
            humanoid:TakeDamage(math.huge)
            
        else
            return
        end
        else
        warn("Humm... Je crois que tu n'a pas mis de son ni d'image...")
        return
 
    end
end)
