Code #1

local Vitesse = 100
local Delais = 20
local msg1 = "Délais : "
local msg2 = "Le Tsunami Arrive !"
local Hauteur = 50
 
script.Script.Enabled = false
function Tsunami(a, b , c, d, e)
    local msg = Instance.new("Hint")
    msg.Parent = workspace
    for i = 1,b do
        msg.Text = c..b-i
        wait(1)
    end
    msg.Text = d
    wait(2)
    msg:Destroy()
    
    local part = Instance.new("Part")
    local scripts = script.Script:Clone()
    scripts.Parent = part
    scripts.Enabled = true
    part.Parent = workspace
    part.BrickColor = BrickColor.new("Dark blue")
    part.Position = Vector3.new(0,0,1000)
    part.Size = Vector3.new(2048,200,300)
    part.Anchored = false
    part.CanCollide = false
    local Velocity = Instance.new("LinearVelocity", part)
    local Attach = Instance.new("Attachment", part)
    Velocity.Attachment0 = Attach
    Velocity.MaxForce = math.huge
    Velocity.VectorVelocity = part.CFrame.LookVector * a
    script.Parent = part
    
    while wait() do
 
        if part.Position.Z <= Vector3.new(0,0,-1000).Z then
            script.Parent = workspace
            part:Destroy()
            Tsunami(Vitesse, Delais, msg1, msg2, Hauteur)
            break
        end
        wait()
    end
end
 
 
Tsunami(Vitesse, Delais, msg1, msg2, Hauteur)
