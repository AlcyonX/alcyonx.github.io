Comment créer une invasion de zombies ?

Vidéo

Code

local NombreParRaid = 50
local Delais = 30
local TempsdunePartie = 120
local msg1 = "Délais : "
local msg2 = "Le Raid a commencé !"
local msg3 = "Le Raid est Terminé !"
local spawns = Vector3.new(150,5,0)
 
function Raid(a, b ,c, d, e, f, g)
    local msg = Instance.new("Hint")
    msg.Parent = workspace
    for i = 1,b do
        msg.Text = c..b-i
        wait(1)
    end
    msg.Text = msg2
    wait(2)
    local Zombiem = game.ReplicatedStorage:FindFirstChildOfClass("Model")
    if Zombiem == nil then
        warn("Attention ! Tu n'as pas mis un Zombie dans 'ReplicatedStorage' !")
        return
    end
    local Zombie = Zombiem:FindFirstChild("Humanoid")
    if Zombie == nil then
        warn("Attention ! Ton Zombie n'est pas un 'Humanoid' !")
        return
    end
 
    for i = 1,a do
        local Zombies = Zombiem:Clone()
        Zombies.Parent = workspace
        Zombies:MoveTo(g)
    end
    for i = 1,e do
        msg.Text = c..e-i
        wait(1)
    end
    
    local work = game.Workspace:GetChildren()
    for i = 1,#work do
        if work[i].Name == Zombiem.Name and work[i]:FindFirstChild("Humanoid") ~= nil then
            work[i]:Destroy()
        end
    end
    
    msg.Text = f
    wait(2)
    msg:Destroy()
    Raid(NombreParRaid, Delais, msg1, msg2, TempsdunePartie, msg3, spawns)
    
end
 
Raid(NombreParRaid, Delais, msg1, msg2, TempsdunePartie, msg3, spawns)
