Comment spawn un vehicule ?

Vidéo

Code

local Temps_pour_respawn = 5
 
local vehicule = script.Parent:FindFirstChild("Vehicule")
vehicule.Parent = game.ReplicatedStorage
script.Parent.Anchored = true
                 
local de = false
script.Parent.BrickColor = BrickColor.new("Lime green")
script.Parent.Touched:Connect(function(hit)
    local plr
    local success,message = pcall(function()
        plr = game.Players:GetPlayerFromCharacter(hit.Parent)
    end)
    if success and plr ~= nil and de == false then
        de = true
        script.Parent.BrickColor = BrickColor.new("Really red")
        local clone = vehicule:Clone()
        clone.Parent = workspace
        wait(Temps_pour_respawn)
        script.Parent.BrickColor = BrickColor.new("Lime green")
        de = false        
    end
end)
