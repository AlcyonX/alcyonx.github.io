Code

local Seat = script.Parent
local Attachement = Instance.new("Attachment", Seat)
Attachement.Orientation = Vector3.new(0,90,0)
Attachement.Axis = Vector3.new(0,0,-1)
local AngularVelocity = Instance.new("AngularVelocity",script)
AngularVelocity.Attachment0 = Attachement
AngularVelocity.MaxTorque = 10000
AngularVelocity.AngularVelocity = Vector3.new(0,0,0)
AngularVelocity.Enabled = true
local LinearVelocity = Instance.new("LinearVelocity",script)
LinearVelocity.Attachment0 = Attachement
LinearVelocity.VelocityConstraintMode = Enum.VelocityConstraintMode.Line
LinearVelocity.MaxForce = 10000
LinearVelocity.LineVelocity = 0
LinearVelocity.RelativeTo = Enum.ActuatorRelativeTo.Attachment0
 
 
local params = RaycastParams.new()
params.FilterDescendantsInstances = {workspace.Terrain}
params.FilterType = Enum.RaycastFilterType.Whitelist
 
 
Seat.Changed:Connect(function()
    script.AngularVelocity.AngularVelocity = Vector3.new(0, -1 * Seat.Steer, 0)
    script.LinearVelocity.LineVelocity = 50 * Seat.Throttle
 
 
 
 
    while task.wait() do
        local sPos = Seat.Position + Vector3.new(0, 5, 0)
        local ePos = Seat.Position - Vector3.new(0, 15, 0)
        local ray = workspace:Raycast(sPos,ePos - sPos,params)
 
        if ray and ray.Material == Enum.Material.Water then
            script.AngularVelocity.Enabled = true
            script.LinearVelocity.Enabled = true
        else
            script.AngularVelocity.Enabled = false
            script.LinearVelocity.Enabled = false
        end
    end
end)
