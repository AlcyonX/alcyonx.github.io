Comment faire un double saut ?

Vidéo

Code

local SautMax = 2
local Delais = 0.2
 
 
local UIS = game:GetService("UserInputService")
local Player = game.Players.LocalPlayer
local Character = Player.Character or Player.CharacterAdded:Wait()
local Humanoid = Character:WaitForChild("Humanoid")
local NumJumps = 0
local canjump = false
 
--Fonction
Humanoid.StateChanged:Connect(function(a, b)
    if Enum.HumanoidStateType.Landed == b then
        NumJumps = 0
        canjump = false
    elseif Enum.HumanoidStateType.Freefall == b then
        wait(Delais)
        canjump = true
    elseif Enum.HumanoidStateType.Jumping == b then
        canjump = false
        NumJumps += 1
    end
end)
 
UIS.JumpRequest:Connect(function()
    if canjump and NumJumps < SautMax then
        Humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
    end
end)
