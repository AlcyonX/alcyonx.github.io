Comment créer une playlist de musique ?

Vidéo

Code

while true do
    local Sounds = script:GetChildren()
    local RandomIndex = math.random(1,#Sounds)
    local RandomSound = Sounds[RandomIndex]
    RandomSound:Play()
    RandomSound.Ended:Wait()
end
