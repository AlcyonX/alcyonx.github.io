local Vitesse = 50
local Delais = 120
local msg1 = "Délais : "
local msg2 = "Le Sol devient de la Lave !"
local Hauteur = 200
 
--Fonctions
 
script.Script.Enabled = false
 
 
function Lava(a, b , c, d, e)
    local msg = Instance.new("Hint")
    msg.Parent = workspace
    for i = 1,b do
        msg.Text = c..b-i
        wait(1)
    end
    msg.Text = d
    wait(2)
    msg:Destroy()
 
    local part = Instance.new("Part")
    local scripts = script.Script:Clone()
    scripts.Parent = part
    scripts.Enabled = true
    part.Parent = workspace
    part.BrickColor = BrickColor.new("Really red")
    part.Material = Enum.Material.CrackedLava
    part.Position = Vector3.new(0,-100,0)
    part.Size = Vector3.new(2048,e,2048)
    part.Anchored = false
    part.CanCollide = false
    local Velocity = Instance.new("LinearVelocity", part)
    local Attach = Instance.new("Attachment", part)
    Velocity.Attachment0 = Attach
    Velocity.MaxForce = math.huge
    Velocity.VectorVelocity = Vector3.new(0,a,0)
    script.Parent = part
 
    while wait() do
 
        if part.Position.Y >= Vector3.new(0,1400,0).Y then
            Velocity.VectorVelocity = Vector3.new(0,0,0)
            wait(4)
            script.Parent = workspace
            part:Destroy()
            Lava(Vitesse, Delais, msg1, msg2, Hauteur)
            break
 
        end
        wait()
    end
end
 
 
Lava(Vitesse, Delais, msg1, msg2, Hauteur)
