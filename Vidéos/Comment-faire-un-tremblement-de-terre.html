function EarthQuake()
	local work = workspace:GetDescendants()
	for i = 1, #work do
		if work[i]:isA("Part") and work[i].Name ~= "Baseplate" and work[i]:isA("SpawnLocation") == false then
			local killscript = script.Script:Clone()
			killscript.Parent = work[i]
			killscript.Enabled = true
			work[i].Anchored = false
			wait()
		elseif work[i]:isA("Snap") or work[i]:isA("Weld") then
			work[i]:Destroy()
		elseif work[i]:isA("SpawnLocation") then
			work[i].CanCollide = false
		end
	end
end
EarthQuake()

local degat = 5 
local AntiLag = 20
local de = false

script.Parent.Touched:Connect(function(hit)
    local H = hit.Parent:FindFirstChild("Humanoid")
    if H ~= nil and de == false then
        de = true
        H:TakeDamage(degat)
        wait(3)
        de = false
    end
end)
 
wait(AntiLag)
script.Parent:Destroy()
